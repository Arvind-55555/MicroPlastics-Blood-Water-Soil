# Commands to deploy GitHub Pages
# Run these commands in order:

# 1. Install git (if not installed)
sudo apt update && sudo apt install -y git

# 2. Navigate to project directory
cd /home/arvind/Downloads/Projects

# 3. Initialize git (if not already)
git init

# 4. Configure git (if needed)
git config user.name "Arvind-55555"
git config user.email "your-email@example.com"

# 5. Add remote repository
git remote add origin git@github.com:Arvind-55555/MicroPlastics-Blood-Water-Soil.git
# OR if remote exists, update it:
# git remote set-url origin git@github.com:Arvind-55555/MicroPlastics-Blood-Water-Soil.git

# 6. Add all files
git add .

# 7. Commit
git commit -m "Deploy Microplastic Detection ML Pipeline with GitHub Pages dashboard"

# 8. Push to main branch
git branch -M main
git push -u origin main

# 9. Enable GitHub Pages:
#    - Go to: https://github.com/Arvind-55555/MicroPlastics-Blood-Water-Soil/settings/pages
#    - Source: Deploy from a branch
#    - Branch: main
#    - Folder: /docs
#    - Click Save

# Dashboard will be available at:
# https://arvind-55555.github.io/MicroPlastics-Blood-Water-Soil
